       ```python
       def cmd_restore():
           """Restore cli.js from pristine backup."""
           cli_dir = get_cli_dir()  # 来自 cli.py
           bm = BackupManager(cli_dir)
           result = bm.restore()
           if not result['ok']:
               # 如果 hash_mismatch，自动重新创建备份
               if result.get('error') == 'hash_mismatch':
                   create_result = bm.ensure_backup()
                   if create_result['ok']:
                       result = bm.restore()
                       if result['ok']:
                           output_json(result)
                           return
               output_error(result.get('error', 'unknown'), result.get('hint', ''))
           else:
               # restore 后验证 cli.js 纯净性 (BAK-07)
               if not bm._is_pristine_file(bm.cli_js):
                   output_error("restore verification failed", "cli.js still contains CJK after restore")
               output_json(result)
       ```
