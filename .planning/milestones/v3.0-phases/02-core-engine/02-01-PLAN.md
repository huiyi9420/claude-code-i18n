       Pattern (from RESEARCH.md scanner example):
       ```python
       CJK_PATTERN = re.compile(r'[\u4e00-\u9fff]')

       def scan_candidates(content, existing, skipped, noise_re):
           candidates = re.findall(r'"([A-Z][A-Za-z][^"]{4,120})"', content)
           results = []
           seen = set()
           for s in candidates:
               if s in seen: continue
               seen.add(s)
               if s in existing or s in skipped: continue
               if " " not in s: continue
               if any(c in s for c in ["=>", "${", "===", "!==", "async ", "function ", ".prototype.", "()"]): continue
               sl = s.lower()
               if "http://" in sl or "https://" in sl: continue
               if s == s.upper() and len(s) > 5: continue
               if noise_re.search(s): continue
               if CJK_PATTERN.search(s): continue  # EXTRACT-06
               # Check for underscores without natural language punctuation
               if '_' in s and not any(c in s for c in ' .,;:!?'): continue
               count = content.count(f'"{s}"')
               result = score_candidate(s, count)
               if result["type"] == "none": continue
               results.append({"en": s, "count": count, "score": result["score"], "type": result["type"]})
           results.sort(key=lambda x: x["score"], reverse=True)
           return results
       ```
